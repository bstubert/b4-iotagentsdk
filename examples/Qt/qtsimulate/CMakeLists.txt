project(example_Qt)

set(CMAKE_INCLUDE_CURRENT_DIR ON)
set(CMAKE_AUTOMOC ON)
set(CMAKE_AUTORCC ON)
set(CMAKE_AUTOUIC ON)

find_package(Qt${QT_MAJOR_VERSION} COMPONENTS Core Gui Quick REQUIRED)

add_executable(qt_simulate
	"main.cpp"
	"AppModel.h"
	"AppModel.cpp"
	"qml.qrc"
	"QRCodeImageProvider.h"
	"QRCodeImageProvider.cpp"
	"ThirdParty/qrcodegen.hpp"
	"ThirdParty/qrcodegen.cpp"
)

target_include_directories(qt_simulate PRIVATE ../../../TVQtRC/Library/export)
target_link_libraries(qt_simulate Qt${QT_MAJOR_VERSION}::Core Qt${QT_MAJOR_VERSION}::Gui Qt${QT_MAJOR_VERSION}::Quick)

install(TARGETS qt_simulate
	EXPORT ${TV_PACKAGE_NAME}
	RUNTIME DESTINATION ${CMAKE_INSTALL_BINDIR})
